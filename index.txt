        function hideMenu() {
            const menu = document.getElementById('mobileMenu');
            // Only hide if the user is on a desktop screen (where onmouseleave is practical)

            function populateStates() {
                const sortedStates = Object.keys(statesData).sort();
                stateSelect.innerHTML = '<option value="">-- Select State / UT --</option>';
                sortedStates.forEach(stateName => {
                    const option = document.createElement('option'); option.value = stateName; option.textContent = stateName; stateSelect.appendChild(option);
                });
            }

            function populateDistricts() {
                const selectedState = stateSelect.value;
                districtSelect.innerHTML = '<option value="">-- Select District --</option>';
                if (selectedState && statesData[selectedState]) {
                    const districts = statesData[selectedState].districts;
                    if (districts && districts.length > 0) {
                        districts.slice().sort().forEach(district => {
                            const option = document.createElement('option'); option.value = district; option.textContent = district; districtSelect.appendChild(option);
                        });
                        districtSelect.disabled = false;
                    } else {
                        districtSelect.disabled = true;
                        districtSelect.innerHTML = '<option value="">-- No districts found for this state --</option>';
                    }
                } else {
                    districtSelect.disabled = true;
                    districtSelect.innerHTML = '<option value="">-- Select State First --</option>';
                }
            }
            stateSelect.addEventListener('change', populateDistricts);
            populateStates();
            
            showDeviceDetails();
        });

        // Function to handle form submission via Email or WhatsApp after validation
        function handleSubmission(type) {
            const form = document.getElementById('queryForm');
            
            // 1. Validate the form using the browser's built-in check
            if (!form.reportValidity()) {
                // Validation failed (browser shows error messages next to invalid fields)
                return;
            }

            // 2. Gather form data
            const name = document.getElementById('name').value;
            const email = document.getElementById('email').value;
            const phone = document.getElementById('phone').value;
            const queryType = document.getElementById('query_type').value;
            const state = document.getElementById('state').value;
            const district = document.getElementById('district').value;
            const area = document.getElementById('area').value;
            const pincode = document.getElementById('pincode').value;
            const message = document.getElementById('query_message').value;

            const bodyText = `
*-- RoutEye Query --*
Query Type: ${queryType}
Name: ${name}
Email: ${email}
Phone: ${phone}
State: ${state}
District: ${district}
Area: ${area}
Pincode: ${pincode}

Message:
${message}
`;

            if (type === 'email') {
                // Construct mailto link
                const subject = encodeURIComponent(`Query from ${name} (RoutEye Website)`);
                const body = encodeURIComponent(bodyText);
                const mailtoLink = `mailto:routeyes@gmail.com?subject=${subject}&body=${body}`;
                window.open(mailtoLink, '_self');
            } else if (type === 'whatsapp') {
                // Construct WhatsApp link
                const whatsappText = encodeURIComponent(`Hello TracKro, I have a query about your GPS services. My details are:\n\n${bodyText}`);
                const whatsappLink = `https://wa.me/918818869279?text=${whatsappText}`;
                window.open(whatsappLink, '_blank');
            }
        }

        function updateCalculatorInputs() {
            const mode = document.getElementById('calculation-type').value;
            const inp1 = document.getElementById('calc-input-1');
            const inp2 = document.getElementById('calc-input-2');
            const resultDiv = document.getElementById('result');
            if (resultDiv) { resultDiv.style.display = 'none'; resultDiv.textContent = ''; }
            if (mode === 'mileage') {
                inp1.placeholder = 'Enter distance (Km)';
                inp2.placeholder = 'Enter fuel (Liters)';
            } else if (mode === 'distance') {
                inp1.placeholder = 'Enter fuel (Liters)';
                inp2.placeholder = 'Enter mileage (Km/L)';
            } else if (mode === 'fuel') {
                inp1.placeholder = 'Enter distance (Km)';
                inp2.placeholder = 'Enter mileage (Km/L)';
            }
        }

        function calcSmart() {
            const mode = document.getElementById('calculation-type').value;
            const a = parseFloat(document.getElementById('calc-input-1').value);
            const b = parseFloat(document.getElementById('calc-input-2').value);
            const resultDiv = document.getElementById('result');
            resultDiv.style.display = 'block';
            resultDiv.style.backgroundColor = 'var(--primary-green)';
            if (isNaN(a) || isNaN(b) || a <= 0 || b <= 0) {
                resultDiv.textContent = 'Please enter positive numbers in both fields.';
                return;
            }
            if (mode === 'mileage') { 
                resultDiv.textContent = `Mileage: ${(a / b).toFixed(2)} Km/L`;
            }
            else if (mode === 'distance') { 
                resultDiv.textContent = `Distance: ${(a * b).toFixed(2)} Km`;
            }
            else if (mode === 'fuel') { 
                resultDiv.textContent = `Fuel Consumed: ${(a / b).toFixed(2)} Liters`;
            }
            else { 
                resultDiv.textContent = 'Select calculation mode.';
            }
        }

        function clearCalc() { 
            document.getElementById('calc-input-1').value = '';
            document.getElementById('calc-input-2').value = ''; 
            const resultDiv = document.getElementById('result'); 
            resultDiv.style.display = 'none'; 
            resultDiv.textContent = '';
        }

        function showDeviceDetails() { 
            const selectedDevice = document.getElementById('device-selector').value; 
            const deviceItems = document.querySelectorAll('.device-details-container .device-item');
            deviceItems.forEach(item => item.classList.remove('active')); 
            if (selectedDevice !== 'none') { 
                const activeDevice = document.getElementById(selectedDevice); 
                if (activeDevice) activeDevice.classList.add('active');
            } 
        }
